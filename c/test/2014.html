#include <stdio.h>

int index(char *str, char ch) {
    int index = 0;
    while (*str != '\0') {
        if (ch == *str) {
            return index + 1;
        }
        index++;
        str++;
    }
    return -1;
}

int main() {
    printf("%d", index("abcc", 'c'));
    return 0;
}

#include <math.h>
#include <stdio.h>
int isPrime(int num) {
    if (num < 2) {
        return 0;
    }
    for (int i = 2; i <= sqrt(num); i++) {
        if (num % i == 0) {
            return 0;
        }
    }
    return 1;
}

int split(int n, int *a, int *b) {
    for (int i = n - 2; i >= 2; i--) {
        if (isPrime(i) && isPrime(n - i)) {
            *a = i;
            *b = n - i;
            return 1;
        }
    }
    return 0;
}

int main() {
    int n, a, b;
    n = 37;
    if (1 == split(n, &a, &b)) {
        printf("%d %d", a, b);
    } else {
        printf("no result");
    }



    return 0;
}

#include <stdio.h>
#include <string.h>

void maxword(char *s, char *t) {
    char *token;
    token = strtok(s, " ");
    while (token != NULL) {
        if (strstr(t, token)) {
            printf("%s\n", token);
        }
        token = strtok(NULL, " ");
    }
}

int main() {
    char s[] = "This is C programming text";
    char t[] = "This is a text for C programming";
    maxword(s, t);
    return 0;
}

# 第五道

#include <stdio.h>
#include <stdlib.h>
typedef struct student {
    long int num;
    char name[10];
    int age;
    char sex;
    char addr[40];
} Student;
int studentLength;
void check(FILE *fp) {
    if (fp == NULL) {
        printf("openning file failed");
        exit(-1);
    }
}

void print_stu(struct student *s, int m, int n) {
    int i = 0;
    while ((s + i)->num != m) {
        i++;
    }
    while (i <= n) {
        printf("%ld, %s, %d, %c, %s", (s + i)->num, (s + i)->name, (s + i)->age,
               (s + i)->sex, (s + i)->addr);
        i++;
    }
}

struct student search(struct student *s, int num) {
    int low = 0, high = studentLength - 1;
    while (low < high) {
        int mid = (high + low) / 2;
        if ((s + mid)->num < num) {
            low = mid;
        } else if ((s + mid)->num > num) {
            high = mid;
        } else {
            return *(s + mid);
        }
    }
    Student a;
    return a;
}

Student *getData(FILE *fp) {
    check(fp);
    fseek(fp, 0, SEEK_END);
    studentLength = ftell(fp) / sizeof(Student);
    Student *s = (Student *)malloc(sizeof(Student));
    fread(s, sizeof(Student), studentLength, fp);
    return s;
}

void sort(Student *s) {
    for (int i = 0; i < studentLength - 1; i++) {
        for (int j = 0; j < studentLength - 1 - i; j++) {
            if ((s + j)->num > (s + j + 1)->num) {
                Student t = *(s + j);
                *(s + j) = *(s + j + 1);
                *(s + j + 1) = t;
            }
        }
    }
}

int main() {
    FILE *fp = fopen("student.dat", "rb");
    Student *s = getData(fp);
    void sort(s);
    fclose(fp);
    return 0;
}
