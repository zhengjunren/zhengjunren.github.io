```c
#include <stdio.h>

struct w {
    char low;
    char high;
};

union u {
    struct w byte;
    short word;
} uu;



int main() {
    uu.word = 0x1234;
    printf("%04x\n", uu.word);
    printf("%02x\n", uu.byte.high);
    printf("%02x\n", uu.byte.low);
    uu.byte.low = 0xff;
    printf("%04x", uu.word);
    return 0;
}
```


```c
int isPrime(int num) {
    if (num < 2) {
        return 0;
    }
    for (int i = 2; i <= sqrt(num); i++) {
        if (num % i == 0) {
            return 0;
        }
    }
    return 1;
}

int split(int n, int *a, int *b) {
    for (int i = n - 2; i >= 2; i--) {
        if (isPrime(i) && isPrime(n - i)) {
            *a = i;
            *b = n - i;
            return 1;
        }
    }
    return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>
typedef struct student {
    long int num;
    char name[10];
    int age;
    char sex;
    char addr[40];
} Student;
int studentLength;
void check(FILE *fp) {
    if (fp == NULL) {
        printf("openning file failed");
        exit(-1);
    }
}

void print_stu(struct student *s, int m, int n) {
    int i = 0;
    while ((s + i)->num != m) {
        i++;
    }
    while (i <= n) {
        printf("%ld, %s, %d, %c, %s", (s + i)->num, (s + i)->name, (s + i)->age,
               (s + i)->sex, (s + i)->addr);
        i++;
    }
}

struct student search(struct student *s, int num) {
    int low = 0, high = studentLength - 1;
    while (low < high) {
        int mid = (high + low) / 2;
        if ((s + mid)->num < num) {
            low = mid;
        } else if ((s + mid)->num > num) {
            high = mid;
        } else {
            return *(s + mid);
        }
    }
    Student a;
    return a;
}

Student *getData(FILE *fp) {
    check(fp);
    fseek(fp, 0, SEEK_END);
    studentLength = ftell(fp) / sizeof(Student);
    Student *s = (Student *)malloc(sizeof(Student));
    fread(s, sizeof(Student), studentLength, fp);
    return s;
}

void sort(Student *s) {
    for (int i = 0; i < studentLength - 1; i++) {
        for (int j = 0; j < studentLength - 1 - i; j++) {
            if ((s + j)->num > (s + j + 1)->num) {
                Student t = *(s + j);
                *(s + j) = *(s + j + 1);
                *(s + j + 1) = t;
            }
        }
    }
}

int main() {
    FILE *fp = fopen("student.dat", "rb");
    Student *s = getData(fp);
    void sort(s);
    fclose(fp);
    return 0;
}
```
