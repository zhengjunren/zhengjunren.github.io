# 选择

  1. 函数在调用过程中的参数被称为（）
    A. 形式参数　　　　B. 函数参数　　　　C. 实际参数　　　　D. 变量参数
    解：函数定义时的参数为：形参；函数调用语句中的参数称为：实参

  2. 在 C 语言中，二位数组元素在内存中的存放顺序是（A）
    A. 按行存放　　　　B. 按列存放　　　　C. 用户自定义　　　　D. 编译器决定

  3. 以下程序的输出结果是（C）

int a = 1, b = 3, c = 0, d = 7, e = 9, f = 2, g = 4, h = 6, i = 8, j = 0,
        k = 10;

    printf("a=%d", a);

# 第五道

#include <stdio.h>
#define N 8

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

void sort(int *num) {
    int maxCount = 0;
    int minCount = N - 1;
    for (int i = 0; i < N / 2; i++) {
        int max = i;
        for (int j = i + 1; j < N - i; j++) {
            if (num[j] > num[max]) {
                max = j;
            }
        }
        swap(&num[maxCount], &num[max]);
        maxCount++;
        int min = N - 1 - i;
        for (int j = N - 1 - i; j > i; j--) {
            if (num[j] < num[min]) {
                min = j;
            }
        }
        swap(&num[minCount], &num[minCount]);
        minCount--;
        if (minCount < maxCount) {
            break;
        }
    }
}

int main() {
    int A[N] = {4, 5, 3, 8, 2, 9, 78, 2};
    sort(A);
    for (int i = 0; i < N; i++) {
        printf("%d ", A[i]);
    }
    return 0;
}

# 第六道

#include <stdio.h>
#include <stdlib.h>
typedef struct node {
    char ch;
    struct node *next;
} Node, *List;

void check(Node *node) {
    if (node == NULL) {
        printf("内存分配失败");
        exit(-1);
    }
}

List create() {
    List head = (Node *)malloc(sizeof(Node));
    check(head);
    head->next = NULL;
    return head;
}

void insert(List list, char ch) {
    while (list->next != NULL) {
        list = list->next;
    }
    Node *node = (Node *)malloc(sizeof(Node));
    check(node);
    node->ch = ch;
    node->next = NULL;
    list->next = node;
}

void output(List head) {
    while (head->next != NULL) {
        head = head->next;
        printf("%c ", head->ch);
    }
}
void sub(List l1, List l2) {
    Node *t = l1;
    // 按照从递增合并入 l1
    while (l1->next != NULL && l2->next != NULL) {
        if (l1->next->ch < l2->next->ch) {
            l1 = l1->next;
        } else {
            Node *t = l2->next;
            l2->next = t->next;
            t->next = l1->next;
            l1->next = t;
        }
    }
    // 如果 l1 到了末尾，而退出循环，
    // 说明 l2 的值均比 l1 的大，
    while (l1->next == NULL) {
        l1->next = l2->next;
    }
    // 翻转 l1
    Node *newhead = t;
    Node *q = t->next;
    newhead->next = NULL;
    while (q != NULL) {
        Node *p = q;
        q = q->next;

        p->next = newhead->next;
        newhead->next = p;
    }
    l1 = newhead;
}

int main() {
    List head = create();
    insert(head, 'K');
    insert(head, 'P');
    insert(head, 'Q');
    List head1 = create();
    insert(head1, 'E');
    insert(head1, 'F');
    insert(head1, 'G');
    sub(head, head1);
    output(head);
    return 0;
}
