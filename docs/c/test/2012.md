#include <stdio.h>

int sub(char *s1, char *s2) {
    while (*s1 != '\0' && *s2 != '\0') {
        if (*s1 != *s2) {
            return *s1 - *s2;
        }
        s1++;
        s2++;
    }
    return *s1 - *s2;
}

int main() {
    printf("%d", sub("abc", "abcd"));
    return 0;
}

#include <stdio.h>

void sort(int *nums, int size) {
    // int j = 0;
    for (int i = 0; i < size - 1; i++) {
        int min = i;
        for (int j = i + 1; j < size; j++) {
            if (nums[j] < nums[min]) {
                min = j;
            }
        }
        int t = nums[min];
        nums[min] = nums[i];
        nums[i] = t;
    }
}

void sort2(int *nums, int size) {
    int j, i;
    for (i = 0; i < size - 1; i++) {
        int max = 0;
        for (j = 0; j < size - i; j++) {
            if (nums[max] < nums[j]) {
                max = j;
            }
        }
        int t = nums[max];
        nums[max] = nums[j - 1];
        nums[j - 1] = t;
    }
}

int compress(int *nums, int size) {
    int slow = 0, fast = 0, count = 0;
    while (fast < size) {
        if (nums[slow] != nums[fast]) {
            nums[++count] = nums[fast];
            slow++;
        }
        fast++;
    }
    return count + 1;
}

int main() {
    int nums[9] = {4, 2, 7, 10, 4, 7, 1, 6, 2};
    sort2(nums, 9);
    int count = compress(nums, 9);
    for (int i = 0; i < count; i++) {
        printf("%d ", nums[i]);
    }
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
typedef struct node {
    int data;
    struct node *next;
} Node, *List;

void check(Node *node) {
    if (node == NULL) {
        printf("内存分配失败，程序退出");
        exit(-1);
    }
}

List create() {
    List head = (List)malloc(sizeof(Node));
    check(head);
    head->next = NULL;
    return head;
}

void insert(List list, int data) {
    while (list->next != NULL) {
        list = list->next;
    }
    Node *node = (Node *)malloc(sizeof(Node));
    check(node);
    node->data = data;
    node->next = list->next;
    list->next = node;
}

void output(List list) {
    while (list->next != NULL) {
        printf("%d ", list->next->data);
        list = list->next;
    }
}

void clear(List list) {
    Node *temp;
    while (list != NULL) {
        temp = list;
        list = list->next;
        free(temp);
    }
}
