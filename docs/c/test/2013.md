```c
        if (*(s + i) != *(s + j)) {
```

```c
#include <stdio.h>

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

void sort(int nums[5], int size) {
    int maxIndex = size - 1;
    int minIndex = 0;
    for (int i = 0; i < (size - 1) / 2; i++) {
        for (int j = minIndex; j < size - 1 - i; j++) {
            if (nums[j + 1] < nums[j]) {
                swap(&nums[j + 1], &nums[j]);
            }
        }
        maxIndex--;
        for (int j = maxIndex; j > minIndex; j--) {
            if (nums[j] < nums[j - 1]) {
                swap(&nums[j], &nums[j - 1]);
            }
        }
        minIndex++;
        if (minIndex >= maxIndex) {
            break;
        }
    }
}

int main() {
    int nums[6] = {5, 2, 3, 7, 0, 11};
    sort(nums, 6);
    for (int i = 0; i < 6; i++) {
        printf("%d ", nums[i]);
    }
    return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>
typedef struct node {
    int data;
    struct node *next;
} Node, *List;

void check(Node *node) {
    if (node == NULL) {
        printf("memory allocation failed!");
        exit(-1);
    }
}

List create() {
    List head = (List)malloc(sizeof(Node));
    check(head);
    head->next = NULL;
    return head;
}

void insert(List list, int data) {
    while (list->next != NULL) {
        list = list->next;
    }
    Node *node = (Node *)malloc(sizeof(Node));
    check(node);
    node->data = data;
    node->next = list->next;
    list->next = node;
}

void output(List list) {
    while (list->next != NULL) {
        printf("%d ", list->next->data);
        list = list->next;
    }
}

int getLength(List list) {
    int len = 0;
    while (list->next != NULL) {
        len++;
        list = list->next;
    }
    return len;
}

Node *getNode(List list, int index) {
    if (index < -1 || index > getLength(list)) {
        return NULL;
    }
    for (int i = -1; i < index; i++) {
        list = list->next;
    }
    return list;
}

void sort(List list) {
    int len = getLength(list);
    for (int i = 0; i < len; i++) {
        int min = i;
        for (int j = i + 1; j < len; j++) {
            if (getNode(list, min)->data > getNode(list, j)->data) {
                min = j;
            }
        }
        Node *beforeMin = getNode(list, min - 1);
        Node *minNode = beforeMin->next;
        beforeMin->next = minNode->next;
        minNode->next = list->next;
        list->next = minNode;
    }
}

int main() {
    List head = create();
    insert(head, 6);
    insert(head, 1);
    insert(head, 3);
    insert(head, 7);
    insert(head, 0);
    sort(head);
    output(head);
    return 0;
}
```
