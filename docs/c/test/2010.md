#include <stdio.h>

#define N 3

void reverse(int (*s)[3]) {
    for (int i = 0; i < N; i++) {
        for (int j = 0; j <= i; j++) {
            int t = *(*(s + j) + i);
            *(*(s + j) + i) = *(*(s + i) + j);
            *(*(s + i) + j) = t;
        }
    }
}

int main() {
    int a[N][N] = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
    reverse(a);
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            printf("%d ", a[i][j]);
        }
        putchar('\n');
    }
    return 0;
}

#include <stdio.h>

void cat(char *s, char *t, int n) {
    int i = 0;
    while (*(s + i) != '\0' && i < n) {
        i++;
    }
    int j = 0;
    while (*(t + j) != '\0' && j < n) {
        *(s + i) = *(t + j);
        j++;
        i++;
    }
    s[i] = '\0';
}

int main() {
    char s[81] = "testabcd";
    char t[10] = "ajds";
    cat(s, t, 5);
    puts(s);
    return 0;
}

void encrypt(char *s, char *fileName) {
    FILE *fp = fopen(fileName, "wb");
    if (fp == NULL) {
        printf("文件打开失败，程序退出！");
        exit(-1);
    }
    while (*s != '\0') {
        if (*s >= 'a' && *s <= 'z') {

        fwrite(s, sizeof(char), 1, fp);
